#include <stdio.h>

    int gcd(int,int);

    printf("input two integers:\n");

int gcd(int a,int b)

{

    int t,c;

    if(a<b)

    {

        t=a;

        a=b;

        b=t;

    }

    c=a%b;

    while(c!=0)

    {

        a=b;

        b=c;

        c=a%b;

    }

    return b;

}
